Welcome. To get a feel for Amber, follow along this quick tutorial.
Step 1: Create your app 
App = SC.Application.create({
  rootElement: $('#tutorial-app')
});
Step 2: Create a model 
App.Person = SC.Object.extend({
  firstName: null,
  lastName: null
});
Step 3: Create an array 
App.people = [];
Step 4: Create a view and append it 
SC.View.create({
  template: SC.Handlebars.compile(
    "People: " +
    "<ul>" +
      "{{#each App.people}}" +
        "<li>{{firstName}} {{lastName}}</li>" +
      "{{/each}}" +
    "</ul>")
}).appendTo(App.rootElement);
Step 5: Add yourself to the array 
var me = App.Person.create({
  firstName: "Yehuda",
  lastName: "Katz"
});
App.people.pushObject(me);
Step 6: Add someone else to the array 
var tom = App.Person.create({
  firstName: "Tom",
  lastName: "Dale"
});
App.people.pushObject(tom);
Step 7: Modify yourself 
me.set({ firstName: 'Peter', lastName: 'Wagenet' });
